The Klef Sonatta Website uses a **modular component architecture** where each component is self-contained with its own HTML, CSS, and JavaScript.

## Component Organization

All shared components live in `shared/components/` with a consistent structure:

```

│   ├── load-basics.html  # Base template with navigation/footer
```

## Component Loading System

The **Load Basics** component provides the foundation for all pages:

### How It Works

<Steps>
  <Step title="Include the Loader">
    Any page can load the base template by including the script:
    ```html
    <script src="../../shared/components/load-basics/load-basics.js"></script>
    ```
  </Step>
  
  <Step title="Automatic Path Detection">
    The loader detects where it's being called from:
    ```javascript
    const scriptUrl = new URL(document.currentScript.src);
    const componentBaseUrl = scriptUrl.href.replace('load-basics.js', '');
    ```
  </Step>
  
  <Step title="Calculate Relative Paths">
    Computes the path difference between current page and component:
    ```javascript
    function calculatePathPrefix() {
      const currentPath = window.location.pathname;
      const pathParts = currentPath.split('/').filter(part => part);
      const depth = pathParts.length;
      return '../'.repeat(depth);
    }
    ```
  </Step>
  
  <Step title="Load and Inject">
    Fetches the component, adjusts all paths, and injects into the page:
    ```javascript
    const response = await fetch(componentUrl);
    let htmlContent = await response.text();
    htmlContent = adjustPaths(htmlContent, pathPrefix);
    injectStyles(styles);
    injectHTML(body);
    executeScripts(scripts);
    ```
  </Step>
</Steps>

## Navigation Component

The navigation system is one of the most complex components:

### Navigation System (`navigation-system.js`)

Provides the core navigation functionality:

<CodeGroup>
```javascript Menu Toggle
function toggleMenu() {
  const isOpen = navbar.classList.toggle('mobile-open');
  
  // iOS-compatible scroll lock
  if (typeof ScrollLock !== 'undefined') {
    if (isOpen) {
      ScrollLock.lock();
    } else {
      ScrollLock.unlock();
    }
  }
  
  // Hide Dynamic Island when menu opens
  if (typeof hideDynamicIsland === 'function') {
    hideDynamicIsland();
  }
}
```

```javascript Mega Menu
function openMegaMenu(selector) {
  const menu = document.querySelector(selector);
  if (!menu) return;
  
  // Close all mega menus
  document.querySelectorAll('.mega-menu')
    .forEach(m => m.classList.remove('active'));
  
  // Show selected menu
  menu.classList.add('active');
  currentMegaMenu = menu;
  
  // Mobile-specific UI
  if (window.innerWidth <= 768) {
    megaTopbar?.classList.add('active');
    backBtn?.classList.add('visible');
  }
}
```

```javascript Event Listeners
function initNavigationListeners() {
  // Toggle menu buttons
  document.querySelectorAll('[data-action="toggle-menu"]')
    .forEach(btn => {
      btn.addEventListener('click', (e) => {
        e.preventDefault();
        toggleMenu();
      });
    });
  
  // Mega menu items (mobile click)
  document.querySelectorAll('[data-mega]').forEach(btn => {
    btn.addEventListener('click', (e) => {
      if (window.innerWidth <= 768) {
        e.preventDefault();
        const selector = btn.getAttribute('data-mega');
        openMegaMenu(selector);
      }
    });
    
    // Desktop hover
    btn.addEventListener('mouseenter', (e) => {
      if (window.innerWidth > 768) {
        const selector = btn.getAttribute('data-mega');
        openMegaMenu(selector);
      }
    });
  });
}
```
</CodeGroup>

### Responsive Behavior

The navigation adapts based on viewport width:

<Tabs>
  <Tab title="Mobile (≤768px)">
    - **Full-screen overlay** with slide-in animation
    - **Mega menu items** open in a separate view with back button
    - **Scroll locking** prevents body scroll when open
    - **Touch-optimized** tap targets
  </Tab>
  
  <Tab title="Desktop (>768px)">
    - **Hover-triggered** mega menus
    - **Dropdown panels** appear below navigation bar
    - **Mouse tracking** closes menus when cursor leaves area

## Dynamic Island Component

An iOS-inspired notification UI that appears at the top of the screen:

```javascript dynamic-island.js
let islandTimeout;
const islandContainer = document.querySelector('.dynamic-island-container');
const islandContent = document.querySelector('.dynamic-island-content');

function showDynamicIsland(message, duration = 3000) {
  if (!islandContainer || !islandContent) return;
  
  // Update content
  islandContent.textContent = message;
  
  // Show island
  islandContainer.classList.add('visible');
  
  // Auto-hide after duration
  clearTimeout(islandTimeout);
  islandTimeout = setTimeout(() => {
    hideDynamicIsland();
  }, duration);
}

function hideDynamicIsland() {
  islandContainer?.classList.remove('visible');
}
```

### Usage

```javascript
// Show a notification
showDynamicIsland('Page saved successfully', 2000);

// Show an error
showDynamicIsland('Failed to load content', 4000);
```

### Sheet System (`sheet-system.js`)

```javascript
function loadSheet(config) {
  const sheet = createSheet(config);
  document.body.appendChild(sheet);
  
  // Trigger opening animation
  requestAnimationFrame(() => {
    sheet.classList.add('open');
    ScrollLock.lock();
  });
  
  return sheet;
}

function createSheet(config) {
  const sheet = document.createElement('div');
  sheet.className = 'adaptive-sheet';
  sheet.innerHTML = `
    <div class="sheet-backdrop"></div>
    <div class="sheet-container">
      <div class="sheet-handle"></div>
      <div class="sheet-content">
        ${config.content || ''}
      </div>
    </div>
  `;
  
  attachSheetHandlers(sheet);
  return sheet;
}
```

### Tab Navigation

Smooth tab switching for portfolio sections:

```javascript tabs-navigation.js
function initTabNavigation() {
  const tabs = document.querySelectorAll('.tab');
  const panels = document.querySelectorAll('.tab-panel');
  
  tabs.forEach(tab => {
    tab.addEventListener('click', (e) => {
      e.preventDefault();
      
      // Update active states
      tabs.forEach(t => t.classList.remove('active'));
      panels.forEach(p => p.classList.remove('active'));
      
      tab.classList.add('active');
      const targetPanel = document.querySelector(tab.dataset.target);
      targetPanel?.classList.add('active');
    });
  });
}
```

## Search Component

Powerful search system with filtering:

### Search System Architecture

<Steps>
  <Step title="Query Parser">
    Parses search queries and extracts filters:
    ```javascript
    parseQuery("design tag:branding type:portfolio")
    // Returns: { terms: ['design'], filters: { tag: 'branding', type: 'portfolio' } }
    ```
  </Step>
  
  <Step title="Match Engine">
    Scores items based on query relevance:
    ```javascript
    calculateScore(item, query)
    // Returns: { score: 0.85, matches: ['title', 'description'] }
    ```
  </Step>
  
  <Step title="Results Rendering">
    Displays filtered and sorted results with highlighting
  </Step>
</Steps>

### Query Parser (`query-parser.js`)

```javascript
const QueryParser = {
  parse(queryString) {
    const filters = {};
    const terms = [];
    
    // Extract filters (tag:value, type:value)
    const filterRegex = /(\w+):(\w+)/g;
    queryString.replace(filterRegex, (match, key, value) => {
      filters[key] = value;
      return '';
    });
    
    // Extract search terms
    const cleanQuery = queryString.replace(filterRegex, '').trim();
    if (cleanQuery) {
      terms.push(...cleanQuery.split(/\s+/));
    }
    
    return { terms, filters };
  }
};
```

## Component Communication

Components communicate through multiple patterns:

### Global Functions

Core functions exposed to global scope:

```javascript
window.toggleMenu = toggleMenu;
window.openMegaMenu = openMegaMenu;
window.showDynamicIsland = showDynamicIsland;
window.loadSheet = loadSheet;
```

### Custom Events

Components dispatch events for loose coupling:

```javascript
// Dispatch event
window.dispatchEvent(new CustomEvent('loadBasicsReady', {
  detail: { timestamp: Date.now() }
}));

// Listen for event
window.addEventListener('loadBasicsReady', (e) => {
  console.log('Load Basics ready at:', e.detail.timestamp);
});
```

### Data Attributes

HTML data attributes trigger behavior:

```html
<!-- Toggle menu -->
<button data-action="toggle-menu">Menu</button>

<!-- Open mega menu -->
<a data-mega="#services-menu">Services</a>

<!-- Open contact sheet -->
<button data-action="open-contact-sheet">Contact Us</button>
```

## Component Lifecycle

Typical component initialization flow:

<Steps>
  <Step title="DOM Ready">
    Wait for DOM to be ready:
    ```javascript
    if (document.readyState === 'loading') {
      document.addEventListener('DOMContentLoaded', init);
    } else {
      init();
    }
    ```
  </Step>
  
  <Step title="Query Elements">
    Find required DOM elements:
    ```javascript
    function init() {
      navbar = document.getElementById('main-nav');
      megaMenus = document.querySelectorAll('.mega-menu');
    }
    ```
  </Step>
  
  <Step title="Attach Listeners">
    Set up event listeners:
    ```javascript
    document.querySelectorAll('[data-action]').forEach(el => {
      el.addEventListener('click', handleAction);
    });
    ```
  </Step>
  
  <Step title="Initial State">
    Set initial component state:
    ```javascript
    closeMenu();
    updateActiveStates();
    ```
  </Step>
</Steps>

## Best Practices

<AccordionGroup>
  <Accordion title="Self-Contained Components">
    Each component manages its own state and doesn't rely on global state:
    ```javascript
    // Good: Component-scoped state
    (function() {
      let isOpen = false;
      let currentItem = null;
      
      function toggle() {
        isOpen = !isOpen;
      }
    })();
    ```
  </Accordion>
  
  <Accordion title="Defensive Initialization">
    Always check if required elements exist:
    ```javascript
    function init() {
      const navbar = document.getElementById('main-nav');
      if (!navbar) {
        console.warn('Navbar not found, skipping initialization');
        return;
      }
      // Continue initialization
    }
    ```
  </Accordion>
  
  <Accordion title="Event Delegation">
    Use event delegation for dynamic content:
    ```javascript
    // Good: Event delegation
    document.addEventListener('click', (e) => {
      const btn = e.target.closest('[data-action]');
      if (btn) handleAction(btn);
    });
    
    // Bad: Direct binding (won't work for dynamic elements)
    document.querySelectorAll('[data-action]').forEach(btn => {
      btn.addEventListener('click', handleAction);
    });
    ```
  </Accordion>
  
  <Accordion title="Cleanup on Destroy">
    Remove event listeners when component is destroyed:
    ```javascript
    function destroy() {
      document.removeEventListener('click', handleClick);
      ScrollLock.unlock();
      currentMegaMenu = null;
    }
    ```
  </Accordion>
</AccordionGroup>
